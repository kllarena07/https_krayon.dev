# Block URL parameters that create duplicate content (tracking/sorting)
Disallow: /*?sort=
Disallow: /*?ref=
Disallow: /*?utm_

# Allow CSS and JS (Crucial for Google to "see" your site layout correctly)
Allow: /wp-admin/admin-ajax.php
Allow: /wp-includes/*.js
